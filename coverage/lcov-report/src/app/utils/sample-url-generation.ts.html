function generateSearchParameters(urlObject: URL, search: string) {
  const searchParameters = urlObject.search;
  if (searchParameters) {
    try {
      search = decodeURI(searchParameters);
    }
    catch (error) {
      if (error.message === 'URI malformed') {
        search = searchParameters;
      }
    }
  }
  return search;
